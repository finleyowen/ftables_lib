/*
FSheets DDL Schema Example 2

Demonstrates some more complex data types in FSheets DDL including nullable
types.

Note type aliases cannot specify the nullability of a type; hence the following  
would be invalid:

type optInt1to5 int<1,5>?;

Instead, define the alias like this:
*/

type int1to5 int<1,5>;

/* Then use it like in Table1 */
table Table1 (
	col1: int1to5?
);

/* 
Note that for the string type, FSheets doesn't differentiate between null 
and empty string. Therefore, all of the columns in Table2 below will be 
equivalent except for col4, the only column that won't accept an empty string
as input 
*/
table Table2 (
	col1: str<,5>,
	col2: str<,5>?,
	col3: str<1,5>?,
	col4: str<1,5>
);

/*
Note that in Table3 below:
- col1 accepts "", "abc", "abcd", "abcde"; it doesn't accept "a" or "ab" or 
  "abcdef"
- col2 accepts "abc", "abcd", "abcde"; it doesn't accept "" or "a" or "ab" or 
  "abcdef"
- col5 accepts "abcde"; it doesn't accept "", "a", ... "abcd", "abcdef"
*/
table Table3 (
	col1: str<3,5>?,
	col2: str<3,5>,
	col3: str<5,5>
);
