/*
	FSheets DDL Schema Example 1

	Demonstrates the FSheets DDL by defining some data types and table schemas.
	
	Complete with comments!
*/

// useful types in the spreadsheet
type uIntType int<0, >;
type userNameType str<2, 32>;
type ratingType double<0.0, 10.0>;

// table storing info about users
table Users (
	userId: uIntType,
	userName: userNameType,
	bio: str?
);

// table storing info about movies
table Movie (
	movieId: uIntType,
	movieName: str,
	yearReleased: uIntType
);

// table storing info about movie ratings 
table MovieRatings (
	userId: uIntType,
	movieId: uIntType,
	rating: ratingType
);
